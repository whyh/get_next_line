char			*ft_strjoin(char const *s1, char const *s2)
{
	char	*str;
	size_t	i;

    printf("6.1|%s|--|%s|\n", s1, s2);
	if (s1 == NULL && s2 == NULL)
		return (NULL);
    printf("6.2|%s|--|%s|\n", s1, s2);
    i = ft_strlen(s1) + ft_strlen(s2);
    printf("6.3|%lu|--|%s|--|%s|\n", i, s1, s2);
    ++i;
    printf("6.4|%lu|--|%s|--|%s|\n", i, s1, s2);
    if (!(str = (char *)malloc((i) * sizeof(char))))
	    return (NULL);
    printf("6.5|%lu|--|%s|--|%s|\n", i, s1, s2);
    ft_strcpy(str, s1);
    printf("6.6|%s|--|%s|\n", s1, s2);
    ft_strcat(str, s2);
    printf("6.7|%s|--|%s|\n", s1, s2);
    return (str);
}


static int		gnl_part2(char **line, ssize_t rbl, char rbuf[BUFF_SIZE + 1])
{
	char	*wbuf;
	char    *tmpp;
	size_t	ln;

    printf("2|%s|\n", *line);
	if (rbl == -1)
		return (-1);
	rbuf[rbl] = '\0';
	ln = ft_linelen(rbuf);
    printf("3|%s|\n", *line);
	if (!(wbuf = ft_strnew(ln)))
        return (-1);
    printf("4|%s|\n", *line);
	ft_strncpy(wbuf, rbuf, ln);
    printf("5|%s|\n", *line);
	tmpp = NULL;
    printf("6|%s|\n", *line);
	if (!(tmpp = ft_strjoin(*line, wbuf)))
		return (-1);
    printf("7|%s|\n", tmpp);
	if (*line)
        ft_strdel(line);
    printf("8|%s|\n", tmpp);
    *line = tmpp;
    printf("9|%s|\n", *line);
	ft_strdel(&wbuf);
    printf("10|%s|\n", *line);
	return (-42);
}

int				get_next_line(const int fd, char **line)
{
	static t_fd	*fdl;
	char		rbuf[BUFF_SIZE + 1];
	ssize_t		rbl;
	int			retflag;
	size_t		ln;

	if (fd < 0)
        return (-1);
	if ((retflag = static_valid_n_pull(&fdl, line, fd)) != -42)
		return (retflag);
	while ((rbl = read(fd, rbuf, BUFF_SIZE)))
	{
        printf("1|%s|\n", *line);
        if ((retflag = gnl_part2(line, rbl, rbuf)) != -42)
			return (retflag);
        printf("11|%s|\n", *line);
		if ((ssize_t)(ln = ft_linelen(rbuf)) != rbl)
		{
            printf("12|%s|\n", *line);
			if ((ssize_t)ln + 1 != rbl)
            	if (!(fdl = static_store(rbuf + ln + 1, fd, fdl)))
               	 return (-1);
			return (1);
		}
        printf("13|%s|\n", *line);
    }
	if (*line)
        return (1);
	return (0);
}
